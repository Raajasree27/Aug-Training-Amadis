using System;

public class Animal
{
    public void Eat()
    {
        Console.WriteLine("Animal is eating...");
    }
}

// Derived class
public class Dog : Animal
{
    public void Bark()
    {
        Console.WriteLine("Dog is barking...");
    }
}

class Program
{
    static void Main(string[] args)
    {
        Dog mydog = new Dog();
        mydog.Bark();
        mydog.Eat();
    }
}
